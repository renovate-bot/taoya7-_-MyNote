### #使用

**全局组件**

```html

<Tao></Tao>


<script>    
    //注册全局组件
    Vue.component("Tao",{
        template: '<h1>Yo</h1>'
    });


    var app = new Vue({
        el: '#app',
        data: {

        }

    })
</script>
```


```html
<script>  
    //创建构造器
    var Alert = Vue.extend( {
        template:'<button @click="cle">点击</button>',
        methods:{
            cle:function () {
                alert("Yo");
            }
        }
    });
    //注册全局组件
    Vue.component("tao",Alert);
</script>  
```

**局部组件**

```html
<script>
    var show = Vue.extend({
        template:"<h1>Show!</h1>"
    });


    var app = new Vue({
        el: '#app',
        data: {

        },
        components:{
            "show":show
        }

    })
</script>
```

**父子组件**

```html
<script>
    //子
    let son1 = Vue.extend({
        template:"<h1>Son1</h1>"
    });

    let son2 = Vue.extend({
        template:"<h1>Son2</h1>"
    });
    //父
    Vue.component("parent",{

        components:{
            "child1":son1,
            "child2":son2
        },
        template:"<div><child1></child1><child2></child2></div>",
    });


    var app = new Vue({
        el: '#app',
        data: {

        }

    })
</script>
```


### #组件通信

```html
    <info city="北京"></info>
    <info city="南昌"></info>
    <info city="运城"></info>

<template id="info">

        <h1>你所在的城市</h1><h1>{{city}}</h1>

    Vue.component("info",{
        template:"#info",
        props:['city']

        }

    })
</script>
```
